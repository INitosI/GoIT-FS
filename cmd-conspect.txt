pwd - узнать путь
ls - посмотреть текущее содержимое
cd путь - перемещение
cd ~ — повертає в кореневу директорію
cd.. - на 1 назад

git clone ссылка-на-репозиторий - копирует реп с гитхаба
git status - показывает не синхронизированные изменения и файлы, которые не отслеживаються
git add - добавить файл в отслеживаемый
git restore --staged secret.md вилучить зміни, пов'язані з файлом secret.md, з-під контролю, але збереже всі інші зміни у проєкті.

файл .gitignore - позволяет игнорировать файлы внутри списка
Пример:
    Ігнорувати всі .txt файли
    *.txt

    Ігнорувати конкретний файл
    my-passwords.txt

    Ігнорувати всі файли в певній папці
    drafts/*

    Ігнорувати всі .txt файли, крім plans.txt
    *.txt
    !plans.txt

git commit -m "commit message" - локально создать коммит, не синхронизирует с гитом
git log - історія комітів
git push - отправить на гитхаб
